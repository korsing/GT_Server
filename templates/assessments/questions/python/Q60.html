40. 다음은 10진수의 정수를 입력받아 2진수로 변환하는 프로그램의 일부이다. 빈 칸에 들어갈 숫자의 합을 나타낸 보기를 고르시오. 단, 연산기호는 %와 /만 사용하시오. 

                
def A(N):
    if (N == 1):
        return '1'
    R = ''
    while True:
        if (N % 2 == [ 빈 칸 ]):
            R += '0'
        else:
            R += '[ 빈 칸 ]'
        N = N / 2

        if (N == 1):
            R += '1'
            return R[::[ 빈 칸 ]]

N = int(input())
print (A(N))
