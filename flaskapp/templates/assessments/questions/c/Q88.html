48.다음 프로그램을 실행했을 때, 출력 결과가 다음과 같다.
0 2 1 0 0 0 1 2
빈칸에 들어가야 할 숫자들을 순서대로 쓰시오. (단, 숫자들은 1 또는 2이다.)

                
#include 
int main(void){
	int a[8] = { (1), 2, (2), 1, 1, 2, 1, 2 };
	int b[8] = { 1, 2, 1, 2, 2, 1, 1, 2 };
	int num[8] = { -1, -1, -1, -1, -1, -1, -1, -1};
	int i;
	for ( i=0 ; i<8 ; i++){
		if ( (a[i] == 2) && (b[i] == 2) )
			num[i] = 2;
		else if ( (a[i] == 1) && (b[i] == 1) )
			num[i] = 1;
		else
			num[i] = 0;
	}
	for ( i=0 ; i<8 ; i++)
		printf("%d ",num[i]);
	return 0;
}
